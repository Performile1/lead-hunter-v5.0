```sql
CREATE TABLE sni_codes (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    code VARCHAR(10) NOT NULL UNIQUE,
    description TEXT NOT NULL,
    dhl_relevance VARCHAR(20) CHECK (dhl_relevance IN ('critical', 'high', 'medium', 'low')),
    recommended_segment VARCHAR(10),
    priority INTEGER DEFAULT 0,
    active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_sni_codes_relevance ON sni_codes(dhl_relevance);
CREATE INDEX idx_sni_codes_active ON sni_codes(active);
```
